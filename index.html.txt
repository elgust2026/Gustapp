    <script>
        let state = {
            profile: JSON.parse(localStorage.getItem('recomp_p')) || {
                name: "Usuario", weight: 75, height: 175, age: 25, 
                goal: 'recomp', activity: 1.55, apiKey: "",
                targets: { cal: 2100, prot: 165, carb: 200, fat: 60 }
            },
            meals: JSON.parse(localStorage.getItem('recomp_m')) || [],
            currentScan: null
        };

        function updateUI() {
            const totals = state.meals.reduce((acc, m) => ({
                cal: acc.cal + m.cal, prot: acc.prot + m.prot, 
                carb: acc.carb + m.carb, fat: acc.fat + m.fat
            }), { cal: 0, prot: 0, carb: 0, fat: 0 });

            document.getElementById('display-name').innerText = state.profile.name;
            document.getElementById('display-goal').innerText = state.profile.goal === 'recomp' ? 'Recomposición' : (state.profile.goal === 'loss' ? 'Pérdida de Grasa' : 'Ganancia Muscular');
            document.getElementById('avatar').innerText = state.profile.name.substring(0,2).toUpperCase();

            const remCal = Math.max(0, state.profile.targets.cal - totals.cal);
            document.getElementById('cal-remaining').innerText = remCal;
            const progress = Math.min(100, (totals.cal / state.profile.targets.cal) * 100);
            document.getElementById('progress-circle').style.strokeDashoffset = 219.9 - (219.9 * progress / 100);
            document.getElementById('progress-percent').innerText = Math.round(progress) + '%';
            document.getElementById('progress-circle').style.stroke = totals.cal > state.profile.targets.cal ? '#B3261E' : '#6750A4';

            updateBar('prot', totals.prot, state.profile.targets.prot);
            updateBar('carb', totals.carb, state.profile.targets.carb);
            updateBar('fat', totals.fat, state.profile.targets.fat, true);

            const alertBanner = document.getElementById('alert-banner');
            if (totals.fat > state.profile.targets.fat || totals.cal > state.profile.targets.cal) {
                alertBanner.classList.remove('hidden');
                document.getElementById('alert-text').innerText = totals.fat > state.profile.targets.fat 
                    ? `¡Exceso de GRASAS! (+${totals.fat - state.profile.targets.fat}g).`
                    : `¡ कैलोरी Calderas al máximo! Límite superado.`;
            } else alertBanner.classList.add('hidden');

            renderLogs();
        }

        function updateBar(id, cur, target, isFat = false) {
            document.getElementById(`${id}-current`).innerText = cur;
            document.getElementById(`${id}-target`).innerText = target;
            const percent = Math.min(100, (cur / target) * 100);
            const bar = document.getElementById(`${id}-bar`);
            bar.style.width = percent + '%';
            if (isFat && cur > target) bar.classList.replace('bg-[#6750A4]', 'bg-[#B3261E]');
            else bar.classList.replace('bg-[#B3261E]', 'bg-[#6750A4]');
        }

        function renderLogs() {
            const container = document.getElementById('logs-container');
            container.innerHTML = state.meals.length === 0 ? '<p class="text-center text-xs py-10 opacity-50">Sin comidas hoy.</p>' :
            state.meals.map((m, i) => `
                <div class="bg-white border rounded-2xl p-4 flex gap-4">
                    <div class="w-10 h-10 bg-[#F3EDF7] rounded-lg flex items-center justify-center">🍽️</div>
                    <div class="flex-1">
                        <p class="font-bold text-sm">${m.name}</p>
                        <p class="text-[10px] opacity-60">${m.cal} kcal | ${m.prot}g P</p>
                    </div>
                    <button onclick="removeMeal(${i})" class="material-symbols-outlined text-[#B3261E] opacity-30">delete</button>
                </div>
            `).reverse().join('');
        }

        function calculateTargets() {
            const p = state.profile;
            let bmr = (10 * p.weight) + (6.25 * p.height) - (5 * p.age) + 5; 
            let tdee = bmr * p.activity;
            let calGoal = Math.round(p.goal === 'recomp' ? tdee - 150 : (p.goal === 'loss' ? tdee - 450 : tdee + 300));
            const protGoal = Math.round(2.2 * p.weight);
            const fatGoal = Math.round((calGoal * 0.25) / 9);
            const carbGoal = Math.round((calGoal - (protGoal * 4) - (fatGoal * 9)) / 4);
            state.profile.targets = { cal: calGoal, prot: protGoal, carb: carbGoal, fat: fatGoal };
            localStorage.setItem('recomp_p', JSON.stringify(state.profile));
        }

        async function runAiAnalysis() {
            const input = document.getElementById('ai-input').value;
            if (!input) return;
            document.getElementById('loading-ai').classList.remove('hidden');
            document.getElementById('analyze-btn').classList.add('hidden');

            const apiKey = state.profile.apiKey;
            if (!apiKey) {
                setTimeout(() => { showRes({name: input, cal: 400, prot: 30, carb: 40, fat: 12}); }, 1500);
            } else {
                try {
                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        body: JSON.stringify({ contents: [{ parts: [{ text: `Analiza esta comida: "${input}". JSON plano: name, cal, prot, carb, fat.` }] }] })
                    });
                    const data = await response.json();
                    const text = data.candidates[0].content.parts[0].text.replace(/```json|```/g, "").trim();
                    showRes(JSON.parse(text));
                } catch (e) { alert("Error API Key"); closeScan(); }
            }
        }

        function showRes(res) {
            state.currentScan = res;
            document.getElementById('loading-ai').classList.add('hidden');
                document.getElementById('ai-result').classList.remove('hidden');
            document.getElementById('result-name').innerText = res.name;
            document.getElementById('res-cal').innerText = res.cal;
            document.getElementById('res-prot').innerText = res.prot;
            document.getElementById('res-carb').innerText = res.carb;
            document.getElementById('res-fat').innerText = res.fat;
        }

        function confirmMeal() {
            if (state.currentScan) {
                state.meals.push({...state.currentScan, time: new Date().toISOString()});
                localStorage.setItem('recomp_m', JSON.stringify(state.meals));
                closeScan(); updateUI();
            }
        }

        function openScan() { document.getElementById('scan-screen').classList.remove('hidden'); }
        function closeScan() { 
            document.getElementById('scan-screen').classList.add('hidden'); 
            document.getElementById('ai-result').classList.add('hidden');
            document.getElementById('analyze-btn').classList.remove('hidden');
        }
        function openSettings() { document.getElementById('settings-screen').classList.remove('hidden'); }
        function closeSettings() { document.getElementById('settings-screen').classList.add('hidden'); }
        function goHome() { closeSettings(); closeScan(); }
        function removeMeal(i) { state.meals.splice(i, 1); localStorage.setItem('recomp_m', JSON.stringify(state.meals)); updateUI(); }
        function saveProfile() {
            state.profile.name = document.getElementById('user-name').value || "Usuario";
            state.profile.weight = parseFloat(document.getElementById('user-weight').value) || 70;
            state.profile.height = parseFloat(document.getElementById('user-height').value) || 170;
            state.profile.age = parseInt(document.getElementById('user-age').value) || 20;
            state.profile.goal = document.getElementById('user-goal').value;
            state.profile.activity = parseFloat(document.getElementById('user-activity').value);
            state.profile.apiKey = document.getElementById('api-key').value;
            calculateTargets(); updateUI(); closeSettings();
        }

        updateUI();
    </script>
